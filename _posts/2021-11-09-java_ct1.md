## 풀이
~~~java
import java.util.*;
public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		sc.close();
		
		int ber = num;
		int count = 0;
		
		while(true) {
			num = ((num%10)*10) + (((num/10)+(num%10)) % 10);
			count++;
			if(ber == num) {
				break;
			}
		}
		System.out.println(count);
	}
}
~~~
